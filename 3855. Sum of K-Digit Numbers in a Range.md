# Hard

You are given three integers $l$, $r$, and $k$.

Consider all possible integers consisting of exactly $k$ digits, where each digit is chosen independently from the integer range $[l, r]$ (inclusive). If 0 is included in the range, leading zeros are allowed.

Return an integer representing the sum of all such numbers.​​​​​​​ Since the answer may be very large, return it modulo $10^9 + 7$.

```cpp
class Solution
{
private:

    long long mod = 1e9 + 7;

    long long modPow(long long a, long long b)
    {
        long long res = 1;
        a %= mod;

        while (b > 0)
        {
            if (b & 1)
                res = res * a % mod;

            a = a * a % mod;
            b >>= 1;
        }

        return res;
    }

public:

    int sumOfNumbers(int l, int r, int k)
    {
        long long len = r - l + 1;
        long long sum = (1LL * (l + r) % mod) * (len % mod) % mod;
        sum = sum * modPow(2, mod - 2) % mod;  // divide by 2 using inverse

        // len^(k-1)
        long long exp = modPow(len, k - 1);

        // (10^k - 1) / 9
        long long geom = (modPow(10, k) - 1 + mod) % mod;
        geom = geom * modPow(9, mod - 2) % mod;

        long long ans = sum;
        ans = ans * exp % mod;
        ans = ans * geom % mod;

        return ans;
    }
};
```
