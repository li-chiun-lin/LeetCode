# Hard

You are given an undirected graph with $n$ nodes labeled from $0$ to $n - 1$. Node $i$ has a value of $nums[i]$, which is either $0$ or $1$. The edges of the graph are given by a 2D array $edges$ where $edges[i] = [u_i, v_i]$ represents an edge between node $u_i$ and node $v_i$.

For a non-empty subset $s$ of nodes in the graph, we consider the induced subgraph of $s$ generated as follows:

- We keep only the nodes in $s$.
- We keep only the edges whose two endpoints are both in $s$.

Return an integer representing the number of non-empty subsets $s$ of nodes in the graph such that:

- The induced subgraph of $s$ is connected.
- The sum of node values in $s$ is even.

```cpp
class Solution {
    bool check_value(vector<int>& nums, int mask)
    {
        int sum = 0;
        int n = size(nums);

        for (int i = 0; i < n; ++i)
            if ((1 << i) & mask)
                sum += nums[i];

        return sum % 2 == 0;
    }

    int find(vector<int>& g, int x)
    {
        if (x == g[x])
            return x;

        return g[x] = find(g, g[x]);
    }

    void join(vector<int>& g, int x, int y)
    {
        int gx = find(g, x);
        int gy = find(g, y);

        if (gx != gy)
            g[gx] = gy;
    }

    bool check_con(vector<vector<bool>>& con, int mask)
    {
        int n = size(con);
        vector<int> g(n + 1);
        iota(begin(g), end(g), 0);
        vector<int> node;

        for (int i = 0; i < n; ++i)
            if ((1 << i) & mask)
            {
                for (auto j : node)
                    if (con[i][j])
                        join(g, i, j);

                node.push_back(i);
            }

        set<int> ss;

        for (int i : node)
            ss.insert(find(g, i));

        return size(ss) == 1;
    }
    
public:
    int evenSumSubgraphs(vector<int>& nums, vector<vector<int>>& edges) {
        int n = size(nums);
        int ub = 1 << n;
        int ans = 0;
        vector<vector<bool>> con(n, vector<bool>(n));

        for (auto& e : edges)
        {
            con[e[0]][e[1]] = true;
            con[e[1]][e[0]] = true;
        }

        for (int mask = 1; mask < ub; ++ mask)
        {
            if (check_value(nums, mask) && check_con(con, mask))
                ++ ans;
        }

        return ans;
    }
};
```
