# Medium

You are given a string $s$ consisting of lowercase English letters.

A substring is almost-palindromic if it becomes a palindrome after removing exactly one character from it.

Return an integer denoting the length of the longest almost-palindromic substring in $s$.

```cpp
class Solution {
    int count(string& s, int l1, int r1, int len1)
    {
        int n = size(s);

        while (l1 >= 0 && r1 < n && s[l1] == s[r1])
        {
            -- l1;
            ++ r1;
            len1 += 2;
        }

        return len1;
    }

public:
    int almostPalindromic(string s) {
        int n = size(s);
        int ans = min(2, n);

        for (int c = 0; c < n; ++c)
        {
            for (int t = 0; t < 2; ++t)
            {
                int l = c - 1 + t;
                int r = c + 1;
                int len = 1 - t;

                while (l >= 0 && r < n && s[l] == s[r])
                {
                    -- l;
                    ++ r;
                    len += 2;
                }

                ans = max(ans, len);

                if (l >= 0)
                {
                    ans = max(ans, count(s, l - 1, r, len + 1));
                }

                if (r < n)
                {
                    ans = max(ans, count(s, l, r + 1, len + 1));
                }
            }
        }

        return ans;
    }
};
```
