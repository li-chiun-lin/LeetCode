# Medium

Given an integer array $nums$, return the length of the longest strictly increasing subsequence.

```cpp
class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        vector<int> v;
        for(auto &x : nums)
        {
            auto it = lower_bound(v.begin(), v.end(), x);

            if(it == v.end())
                v.push_back(x);
            else
                *it = x;
        }
        
        return v.size();
    }
};
```
