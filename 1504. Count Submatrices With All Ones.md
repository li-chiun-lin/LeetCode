# Medium

Given an $m \times n$ binary matrix $mat$, return the number of submatrices that have all ones.

```cpp
class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int m = size(mat);
        int n = size(mat[0]);
        int ans = 0;

        auto count = [&](int l, int r)
        {
            int c = 0;
            int ub = n;

            for (int i = l; i < m; ++i)
                for (int j = r; j < ub; ++j)
                {
                    if (mat[i][j])
                        ++ c;
                    else
                        ub = j;
                }

            return c;
        };

        for (int i = 0; i < m; ++i)
            for (int j = 0; j < n; ++j)
                ans += count(i, j);

        return ans;
    }
};
```
