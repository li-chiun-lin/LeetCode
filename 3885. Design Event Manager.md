# Medium

You are given an initial list of events, where each event has a unique $eventId$ and a $priority$.

Implement the __EventManager__ class:

- __EventManager(int[][] events)__ Initializes the manager with the given events, where $events[i] = [eventId_i, priority‚Äã‚Äã‚Äã‚Äã‚Äã‚Äã_‚Äãi]$.
- __void updatePriority(int eventId, int newPriority)__ Updates the priority of the active event with id $eventId$ to $newPriority$.
- __int pollHighest()__ Removes and returns the $eventId$ of the active event with the highest priority. If multiple active events have the same priority, return the smallest $eventId$ among them. If there are no active events, return -1.

An event is called active if it has not been removed by __pollHighest()__.

```cpp
class EventManager {
    // Maps Priority -> Set of Event IDs
    map<int, set<int>> priorityToEventIds;
    // Maps Event ID -> Current Priority
    map<int, int> eventIdToPriority;

public:
    EventManager(vector<vector<int>>& events) {
        for (const auto& event : events) {
            int id = event[0];
            int priority = event[1];
            
            priorityToEventIds[priority].insert(id);
            eventIdToPriority[id] = priority;
        }
    }

    void updatePriority(int eventId, int newPriority) {
        int oldPriority = eventIdToPriority[eventId];
        
        // Remove from the old priority group
        priorityToEventIds[oldPriority].erase(eventId);
        if (priorityToEventIds[oldPriority].empty()) {
            priorityToEventIds.erase(oldPriority);
        }

        // Update to the new priority
        eventIdToPriority[eventId] = newPriority;
        priorityToEventIds[newPriority].insert(eventId);
    }

    int pollHighest() {
        if (priorityToEventIds.empty()) {
            return -1;
        }

        // maps are sorted; rbegin() gives us the highest priority
        auto highestPriorityEntry = priorityToEventIds.rbegin();
        int highestPriority = highestPriorityEntry->first;
        
        // Get the first event ID associated with this priority
        int eventIdToReturn = *highestPriorityEntry->second.begin();

        // Cleanup the entry
        highestPriorityEntry->second.erase(eventIdToReturn);
        if (highestPriorityEntry->second.empty()) {
            priorityToEventIds.erase(highestPriority);
        }

        return eventIdToReturn;
    }
};
```
