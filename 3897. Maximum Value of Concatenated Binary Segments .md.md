# Hard

You are given two integer arrays `nums1` and `nums0`, each of size `n`.

- `nums1[i]` represents the number of `'1'`s in the `ith` segment.
- `nums0[i]` represents the number of `'0'`s in the `ith` segment.

For each index `i`, construct a binary segment consisting of:

- `nums1[i]` occurrences of `'1'` followed by
- `nums0[i]` occurrences of `'0'`.

You may **rearrange** the order of these **segments** in any way. After rearranging, **concatenate** all segments to form a single binary string.

Return the **maximum** possible integer value of the concatenated binary string.

Since the result can be very large, return the answer **modulo** $10^9 + 7$.

```cpp
class Solution {
    struct cmp
    {
        bool operator()(pair<int,int> a, pair<int,int> b)
        {
            int lenA = a.first + a.second;
            int lenB = b.first + b.second;
    
            int i = 0, j = 0;
    
            int total = lenA + lenB;
    
            for (int k = 0; k < total; ++k)
            {
                int bitAB, bitBA;
    
                // AB
                if (k < a.first) bitAB = 1;
                else if (k < a.first + a.second) bitAB = 0;
                else if (k < a.first + a.second + b.first) bitAB = 1;
                else bitAB = 0;
    
                // BA
                if (k < b.first) bitBA = 1;
                else if (k < b.first + b.second) bitBA = 0;
                else if (k < b.first + b.second + a.first) bitBA = 1;
                else bitBA = 0;
    
                if (bitAB != bitBA)
                    return bitAB > bitBA;
            }
    
            return false;
        }
    };
    
public:
    int maxValue(vector<int>& nums1, vector<int>& nums0) {
        int n = size(nums1);
        vector<pair<int, int>> nums(n);

        for (int i = 0; i < n; ++i)
            nums[i] = {nums1[i], nums0[i]};

        sort(begin(nums), end(nums), cmp());

        long long val = 0;
        long long mod = 1e9 + 7;

        for (auto [o, z] : nums)
        {
            for (int i = 0; i < o; ++i)
            {
                val = val * 2 + 1;
                val %= mod;
            }

            for (int i = 0; i < z; ++i)
            {
                val = val * 2 % mod;
            }
        }

        return val;
    }
};
```
