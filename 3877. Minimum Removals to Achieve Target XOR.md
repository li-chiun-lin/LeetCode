# Medium

You are given an integer array $nums$ and an integer $target$.

You may remove any number of elements from $nums$ (possibly zero).

Return the minimum number of removals required so that the bitwise XOR of the remaining elements equals $target$. If it is impossible to achieve $target$, return -1.

```cpp
class Solution {
    int dfs(vector<int>& nums, int i, int target, vector<unordered_map<int, int>>& dp)
    {
        if (i == size(nums))
        {
            return target == 0 ? 0 : -1;
        }

        if (dp[i].count(target))
            return dp[i][target];

        int dont = dfs(nums, i + 1, target, dp);
        int take = dfs(nums, i + 1, target ^ nums[i], dp);
        int val = -1;

        if (dont == -1 && take == -1)
            val = -1;
        else if (dont == -1)
            val = take + 1;
        else if (take == -1)
            val = dont;
        else
            val = max(dont, take + 1);

        return dp[i][target] = val;
    }

public:
    int minRemovals(vector<int>& nums, int target) {
        int n = size(nums);
        vector<unordered_map<int, int>> dp(n);
        
        int m = dfs(nums, 0, target, dp);

        return m == -1 ? -1 : n - m;
    }
};
```
