# Medium

Given an integer array $nums$, return the length of the longest subaray that has a bitwise XOR of zero and contains an equal number of even and odd numbers. If no such subarray exists, return $0$.

```cpp
class Solution {
public:
    int maxBalancedSubarray(vector<int>& nums) {
        int n = size(nums);
        int x = 0;
        vector<int> prefix_cnt(n + 1);
        unordered_map<int, unordered_map<int, int>> idx;
        int ans = 0;
        idx[0][0] = 0;

        for (int i = 0; i < n; ++i)
        {
            x ^= nums[i];
            int c = prefix_cnt[i] + (nums[i] % 2 ? 1 : -1);
            
            prefix_cnt[i + 1] = c;

            if (idx.count(x) && idx[x].count(c))
            {
                if (prefix_cnt[idx[x][c]] == c)
                    ans = max(ans, i + 1 - idx[x][c]);
            }
            else
            {
                idx[x][c] = i + 1;
            }
        }

        return ans;
    }
};
```
