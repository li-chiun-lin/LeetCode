# Hard

Given a string of digits $s$, return the number of palindromic subsequences of $s$ having length $5$. Since the answer may be very large, return it modulo $10^9 + 7$.

```cpp
class Solution {
public:
    int countPalindromes(string s) {
        int n = size(s);
        int m = 1e9 + 7;
        int ans = 0;
        vector<vector<vector<int>>> prefix(n + 1, 
            vector<vector<int>>(10, 
            vector<int>(10)));
        auto suffix = prefix;
        vector<int> cnt(10);

        for (int i = 0; i < n; ++i)
        {
            int c = s[i] - '0';

            for (int j = 0; j < 10; ++j)
                for (int k = 0; k < 10; ++k)
                    prefix[i + 1][j][k] = prefix[i][j][k] + (k == c ? cnt[j] : 0);

            ++ cnt[c];
        }

        cnt = vector<int>(10);

        for (int i = n - 1; i >= 0; --i)
        {
            int c = s[i] - '0';

            for (int j = 0; j < 10; ++j)
                for (int k = 0; k < 10; ++k)
                    suffix[i][j][k] = suffix[i + 1][j][k] + (k == c ? cnt[j] : 0);

            ++ cnt[c];
        }

        for (int i = 2; i + 2 < n; ++i)
            for (int j = 0; j < 10; ++j)
                for (int k = 0; k < 10; ++k)
                    ans = (ans + 1LL * prefix[i][j][k] * suffix[i + 1][j][k]) % m;

        return ans;
    }
};
```
