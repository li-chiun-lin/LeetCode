# Easy

Given an integer array $nums$ and an integer $k$, return the smallest positive multiple of $k$ that is missing from $nums$.

```cpp
class Solution {
public:
    int missingMultiple(vector<int>& nums, int k) {
        set<int> ss;

        for (auto x : nums)
            if (x % k == 0)
                ss.insert(x);

        for (int i = 1; i < 200; ++i)
            if (ss.count(k * i) == 0)
                return i * k;

        return -1;
    }
};
```
