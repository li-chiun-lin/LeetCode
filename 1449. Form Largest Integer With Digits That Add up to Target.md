# Hard

Given an array of integers $cost$ and an integer $target$, return the maximum integer you can paint under the following rules:

- The cost of painting a digit $(i + 1)$ is given by $cost[i]$ (0-indexed).
- The total cost used must be equal to $target$.
- The integer does not have $0$ digits.

Since the answer may be very large, return it as a string. If there is no way to paint any integer given the condition, return "0".

```cpp
class Solution {
    bool larger(string& s1, string& s2)
    {
        // if s1 is not okay, 
        if (s1.back() == '0')
            return false;

        // if s2 is not okay
        if (s2.back() == '0')
            return true;
        
        return s1.size() > s2.size();
    }

public:
    string largestNumber(vector<int>& cost, int target) {
        int n = size(cost);
        vector<vector<string>> dp(n, vector<string>(target + 1, "-"));

        function<string(int, int)> dfs = [&](int i, int t)
        {
            // the okay case
            if (t == 0)
                return "";

            // the not okay case
            if (i == n)
                return "0";

            // momorized
            if (dp[i][t] != "-")
                return dp[i][t].c_str();

            auto take = t >= cost[i] ? to_string(i + 1) + dfs(0, t - cost[i]) : "0";
            auto dont = dfs(i + 1, t);

            return (dp[i][t] = larger(take, dont) ? take : dont).c_str();
        };

        return dfs(0, target);
    }
};
```
