# Medium

The beauty of a string is the difference in frequencies between the most frequent and least frequent characters.

Given a string $s$, return the sum of beauty of all of its substrings.

```cpp
class Solution {
public:
    int beautySum(string s) {
        int n = size(s);
        int ans = 0;

        for (int i = 0; i < n; ++i)
        {
            vector<int> cnt(26);
            int ma = 0;
            int mi = 0;

            for (int j = i; j < n; ++j)
            {
                int jj = s[j] - 'a';

                ma = max(ma, ++ cnt[jj]);

                if (mi >= cnt[jj] - 1)
                {
                    mi = cnt[jj];

                    for (int k : cnt)
                        if (k)
                            mi = min(mi, k);
                }

                ans += ma - mi;
            }
        }

        return ans;
    }
};
```
