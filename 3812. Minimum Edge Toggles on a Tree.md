# Hard

You are given an undirected tree with $n$ nodes, numbered from $0$ to $n - 1$. It is represented by a 2D integer array $edges‚Äã‚Äã‚Äã‚Äã‚Äã‚Äã‚Äã$ of length $n - 1$, where $edges[i] = [a_i, b_i]$ indicates that there is an edge between nodes $a_i$ and $b_i$ in the tree.

You are also given two binary strings $start$ and $target$ of length $n$. For each node $x$, $start[x]$ is its initial color and $target[x]$ is its desired color.

In one operation, you may pick an edge with index $i$ and toggle both of its endpoints. That is, if the edge is $[u, v]$, then the colors of nodes $u$ and $v$ each flip from '0' to '1' or from '1' to '0'.

Return an array of edge indices whose operations transform $start$ into $target$. Among all valid sequences with minimum possible length, return the edge indices in increasing‚Äã‚Äã‚Äã‚Äã‚Äã‚Äã‚Äã order.

If it is impossible to transform $start$ into $target$, return an array containing a single element equal to $-1$.

```cpp
class Solution {
public:
    vector<int> minimumFlips(int n, vector<vector<int>>& edges, string start, string target) {
        vector<int> diff(n);

        for (int i = 0; i < n; ++i)
            diff[i] = (start[i] != target[i]);

        vector<vector<pair<int, int>>> adj(n);

        for (int i = 0; i < n - 1; ++i)
        {
            adj[edges[i][0]].push_back({edges[i][1], i});
            adj[edges[i][1]].push_back({edges[i][0], i});
        }

        vector<int> parent(n, -1);
        vector<int> parent_idx(n, -1);
        vector<int> order;
        vector<bool> visited(n);
        queue<int> que;

        que.push(0);
        visited[0] = true;

        while (size(que))
        {
            int u = que.front();
            que.pop();
            order.push_back(u);

            for (auto& [v, i] : adj[u])
                if (! visited[v])
                {
                    visited[v] = true;
                    parent[v] = u;
                    parent_idx[v] = i;
                    que.push(v);
                }
        }

        vector<int> ans;
        for (int i = n - 1; i > 0; --i)
        {
            int u = order[i];
            if (diff[u])
            {
                ans.push_back(parent_idx[u]);
                diff[parent[u]] ^= 1;
            }
        }

        if (diff[0])
            return {-1};

        sort(begin(ans), end(ans));
        return ans;
    }
};
```
