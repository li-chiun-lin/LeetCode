# Medium

You are given an integer $n$.

Let $r$ be the integer formed by reversing the digits of $n$.

Return the sum of all prime numbers between $\min(n, r)$ and $\max(n, r)$, inclusive.

```cpp
class Solution {
    int reverse(int x)
    {
        int y = 0;

        while (x)
        {
            y = y * 10 + (x % 10);
            x /= 10;
        }

        return y;
    }

public:
    int sumOfPrimesInRange(int n) {
        int r = reverse(n);
        int lb = min(r, n);
        int ub = max(r, n);
        vector<bool> sieve(ub + 1, true);
        sieve[0] = sieve[1] = false;

        for (int i = 2; i <= ub; ++i)
            if (sieve[i])
                for (int j = i * i; j <= ub; j += i)
                    sieve[j] = false;

        int cnt = 0;

        for (int i = lb; i <= ub; ++i)
            if (sieve[i])
                cnt += i;

        return cnt;
    }
};
```
