# Hard

Let $f(x)$ be the number of zeroes at the end of $x!$.

Given an integer $k$, return the number of non-negative integers $x$ have the property that $f(x) = k$.

```cpp
class Solution {
public:
    int preimageSizeFZF(int k) {
        int l = 0;
        int r = INT_MAX;
        int m = 0;

        while (l < r)
        {
            m = l + (r - l) / 2;
            int c = 0;
            int mm = m;

            while (m)
            {
                c += m;
                m /= 5;
            }

            if (c < k)
                l = mm + 1;
            else if (c > k)
                r = mm;
            else
                return 5;
        }

        return 0;
    }
};
```
