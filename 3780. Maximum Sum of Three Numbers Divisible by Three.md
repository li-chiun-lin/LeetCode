# Medium

You are given an integer array $nums$.

Your task is to choose exactly three integers from $nums$ such that their sum is divisible by three.

Return the maximum possible sum of such a triplet. If no such triplet exists, return 0.

```cpp
class Solution {
public:
    int maximumSum(vector<int>& nums) {
        vector<vector<int>> vec(3);

        for (int x : nums)
        {
            vec[x % 3].push_back(x);
        }

        int ans = 0;

        for (auto& v : vec)
        {
            sort(rbegin(v), rend(v));

            if (size(v) >= 3)
                ans = max(ans, v[0] + v[1] + v[2]);
        }

        if (size(vec[0]) && size(vec[1]) && size(vec[2]))
            ans = max(ans, vec[0][0] + vec[1][0] + vec[2][0]);

        return ans;
    }
};
```
