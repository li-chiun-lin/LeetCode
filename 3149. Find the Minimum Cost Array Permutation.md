# Hard

You are given an array $nums$ which is a permutation of $[0, 1, 2, ..., n - 1]$. The score of any permutation of $[0, 1, 2, ..., n - 1]$ named $perm$ is defined as:

$score(perm) = |perm[0] - nums[perm[1]]| + |perm[1] - nums[perm[2]]| + ... + |perm[n - 1] - nums[perm[0]]|$

Return the permutation $perm$ which has the minimum possible score. If multiple permutations exist with this score, return the one that is
lexicographically smallest
among them.

```cpp
class Solution {
public:
    vector<int> findPermutation(vector<int>& nums) {
        int n = size(nums);
        int lb = INT_MAX;
        vector<int> ans;
        vector<bool> visited(n);
        vector<int> perm;

        function<void(int)> dfs = [&](int cost)
        {
            if (size(perm) == n)
            {
                int f = cost + abs(perm.back() - nums[perm[0]]);

                if (lb > f)
                {
                    lb = f;
                    ans = perm;
                }

                return ;
            }
            
            for (int i = 0; i < n; ++i)
            {
                if (visited[i])
                    continue;

                if (perm.empty())
                {
                    visited[i] = true;
                    perm.push_back(i);
                    dfs(0);
                    perm.pop_back();
                    visited[i] = false;
                }
                else
                {
                    int c = cost + abs(perm.back() - nums[i]);

                    if (c < lb)
                    {
                        visited[i] = true;
                        perm.push_back(i);
                        dfs(c);
                        perm.pop_back();
                        visited[i] = false;
                    }
                }
            }
        };

        dfs(0);

        return ans;
    }
};
```
