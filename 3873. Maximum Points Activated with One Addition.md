# Hard

You are given a 2D integer array $points$, where $points[i] = [x_i, y_i]$ represents the coordinates of the ith point. All coordinates in $points$ are distinct.

If a point is activated, then all points that have the same x-coordinate or y-coordinate become activated as well.

Activation continues until no additional points can be activated.

You may add one additional point at any integer coordinate $(x, y)$ not already present in $points$. Activation begins by activating this newly added point.

Return an integer denoting the maximum number of points that can be activated, including the newly added point.

```cpp
class Solution {
public:
    int maxActivated(vector<vector<int>>& points) {
        int n = size(points);
        vector<int> g(n);
        vector<int> cnt(n, 1);
        iota(begin(g), end(g), 0);

        if (n <= 2)
            return n + 1;

        function<int(int)> find = [&](int x)
        {
            return x == g[x] ? x : g[x] = find(g[x]);
        };

        auto join = [&](int x, int y)
        {
            int gx = find(x);
            int gy = find(y);

            if (gx != gy)
            {
                g[gx] = gy;
                cnt[gy] += cnt[gx];
                cnt[gx] = 0;
            }

            return gy;
        };

        unordered_map<int, int> xs, ys;

        for (int i = 0; i < n; ++i)
        {
            int x = points[i][0];
            int y = points[i][1];

            if (xs.count(x))
                join(i, xs[x]);
            else
                xs[x] = i;

            if (ys.count(y))
                join(i, ys[y]);
            else
                ys[y] = i;
        }

        sort(rbegin(cnt), rend(cnt));

        return cnt[0] + cnt[1] + 1;
    }
};
```
