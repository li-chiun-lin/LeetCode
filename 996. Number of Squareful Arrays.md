# Hard

An array is squareful if the sum of every pair of adjacent elements is a perfect square.

Given an integer array nums, return the number of permutations of $nums$ that are squareful.

```cpp
class Solution {
    bool check(int a, int b)
    {
        int r = sqrt(a + b);
        return (r * r) == (a + b);
    }

    int dfs(int u, vector<int> nums)
    {
        int n = size(nums);
        int cnt = 0;

        if (u >= n)
            return 1;

        for (int v = u; v < n; ++v)
            if (v == u || nums[v] != nums[u])
            {
                swap(nums[u], nums[v]);

                if (u == 0 || (u > 0 && check(nums[u - 1], nums[u])))
                    cnt += dfs(u + 1, nums);
            }

        return cnt;
    };

public:
    int numSquarefulPerms(vector<int>& nums) {
        sort(begin(nums), end(nums));
        return dfs(0, nums);
    }
};
```

```cpp
class Solution {
    bool check(int a, int b)
    {
        int r = sqrt(a + b);
        return (r * r) == (a + b);
    }

public:
    int numSquarefulPerms(vector<int>& nums) {
        map<int, int> hist;
        map<int, set<int>> cand;

        for (auto x : nums)
            ++ hist[x];

        for (auto& [i, _] : hist)
            for (auto& [j, _] : hist)
                if (check(i, j))
                    cand[i].insert(j);

        function<int(int, int)> dfs = [&](int u, int n)
        {
            if (n == 0)
                return 1;

            -- hist[u];

            int cnt = 0;

            for (int v : cand[u])
                if (hist[v])
                    cnt += dfs(v, n - 1);

            ++ hist[u];

            return cnt;
        };

        int ans = 0;

        for (auto& [u, _] : hist)
            ans += dfs(u, size(nums) - 1);

        return ans;
    }
};
```
