# Medium

You are given an integer $num$.

A number $num$ is called a Complete Prime Number if every prefix and every suffix of $num$ is prime.

Return $true$ if $num$ is a Complete Prime Number, otherwise return $false$.

```cpp
class Solution {
    bool is_prime(int x)
    {
        if (x < 2)
            return false;

        if (x == 2)
            return true;

        if (x % 2 == 0)
            return false;

        for (int i = 3; i * i <= x; i += 2)
            if (x % i == 0)
                return false;

        return true;
    }
    
    bool check_prefix(int x)
    {
        while (x)
        {
            if (! is_prime(x))
                return false;

            x /= 10;
        }

        return true;
    }

    bool check_suffix(int x)
    {
        long long v = 0;
        long long base = 1;

        while (x)
        {
            int d = x % 10;
            v += d * base;
            base *= 10;
            x /= 10;

            if (! is_prime(v))
                return false;
        }

        return true;
    }
    
public:
    bool completePrime(int num) {
        return check_prefix(num) && check_suffix(num);
    }
};
```
