# Medium

You are given a positive integer array $nums$.

For a positive integer $k$, define $nonPositive(nums, k)$ as the minimum number of operations needed to make every element of $nums$ non-positive. In one operation, you can choose an index $i$ and reduce $nums[i]$ by $k$.

Return an integer denoting the minimum value of $k$ such that $nonPositive(nums, k) \leq k^2$.

```cpp
class Solution {
    long long nonPositive(vector<int>& nums, int k)
    {
        long long cnt = 0;

        for (int x : nums)
        {
            if (x >= 0)
            {
                cnt += (x + k - 1) / k;
            }
        }

        return cnt;
    }
    
public:
    int minimumK(vector<int>& nums) {
        int n = size(nums);
        int l = 1;
        int r = INT_MAX;
        int ans = 0;

        while (l <= r)
        {
            long long k = l + (r - l) / 2;
            long long c = nonPositive(nums, k);

            if (c <= k * k)
            {
                ans = k;
                r = k - 1;
            }
            else
            {
                l = k + 1;
            }
        }

        return ans;
    }
};
```
