# Medium

You are given a string $s$ consisting of lowercase English words, each separated by a single space.

Determine how many vowels appear in the first word. Then, reverse each following word that has the same vowel count. Leave all remaining words unchanged.

Return the resulting string.

```cpp
class Solution {
    int count(string& s)
    {
        int cnt = 0;
        string vowel = "aeiou";

        for (char c : s)
            if (vowel.find(c) != string::npos)
                ++ cnt;

        return cnt;
    }

public:
    string reverseWords(string s) {
        istringstream iss(s);
        string t = "";
        string token = "";

        iss >> t;
        int x = count(t);

        while (getline(iss, token, ' '))
        {
            int y = count(token);

            if (y == x)
                reverse(begin(token), end(token));

            t += token;
            t.push_back(' ');
        }

        if (t.back() ==  ' ')
            t.pop_back();

        return t;
    }
};
```
