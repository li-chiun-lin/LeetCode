# Medium

You are given an integer array $nums$.

A mirror pair is a pair of indices $(i, j)$ such that:

- $0 \leq i < j < nums.length$, and
- $reverse(nums[i]) == nums[j]$, where $reverse(x)$ denotes the integer formed by reversing the digits of $x$. Leading zeros are omitted after reversing.

Return the minimum absolute distance between the indices of any mirror pair.

If no mirror pair exists, return $-1$.

```cpp
class Solution {
    int inv(int x)
    {
        int y = 0;

        while (x)
        {
            y = y * 10 + (x % 10);
            x /= 10;
        }

        return y;
    }
    
public:
    int minMirrorPairDistance(vector<int>& nums) {
        int n = size(nums);
        map<int, vector<int>> idx;

        for (int i = 0; i < n; ++i)
            idx[nums[i]].push_back(i);

        for (auto& x : nums)
            x = inv(x);

        int ans = INT_MAX;

        for (int i = 0; i < n; ++i)
        {
            if (idx.count(nums[i]) == 0)
                continue;

            auto& vec = idx[nums[i]];
            auto it = lower_bound(begin(vec), end(vec), i + 1);

            if (it != end(vec))
                ans = min(ans, *it - i);
        }

        return ans == INT_MAX ? -1 : ans;
    }
};
```
