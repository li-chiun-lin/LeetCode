# Medium

You are given an integer array $nums$, and an integer $k$.

For any $nums[l..r]$, define its cost as:

$cost = (\max(nums[l..r]) - \min(nums[l..r])) * (r - l + 1)$.

Return an integer denoting the number of subarrays of $nums$ whose cost is less than or equal to $k$.

```cpp
class Solution {
public:
    long long countSubarrays(vector<int>& nums, long long k) {
        int n = size(nums);
        long long ans = 0;
        deque<int> ma, mi;

        for (int r = 0, l = 0; r < n; ++r)
        {
            while (size(ma) && nums[ma.back()] <= nums[r])
                ma.pop_back();
            ma.push_back(r);

            while (size(mi) && nums[mi.back()] >= nums[r])
                mi.pop_back();
            mi.push_back(r);

            while ((long long)(nums[ma.front()] - nums[mi.front()]) * (r - l + 1) > k)
            {
                if (ma.front() == l)
                    ma.pop_front();
                if (mi.front() == l)
                    mi.pop_front();

                ++ l;
            }

            ans += r - l + 1;
        }

        return ans;
    }
};
```
